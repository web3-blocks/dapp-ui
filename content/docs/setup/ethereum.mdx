To build Ethereum-enabled user experiences in your dApp, **@web3-blocks/dapp-ui** provides:

- Wallet connection & state
- Account management
- Multi-chain support
- Network switching
- Fully typed React hooks

Everything is handled through the **`DappUiProvider`** and simple client components.

---

## Installation

```bash
npm install @web3-blocks/dapp-ui
```

Check out [@web3-blocks/dapp-ui](https://www.npmjs.com/package/@web3-blocks/dapp-ui) documentation

---

## Provider Setup

Wrap your app with the `DappUiProvider` and pass supported chains:

```tsx title="provider/dapp-ui.provider.tsx" showLineNumbers
"use client"

import { DappUiProvider } from "@web3-blocks/dapp-ui"
import { mainnet, sepolia } from "viem/chains"

export function DappUiWrapper({ children }: { children: React.ReactNode }) {
  return (
    <DappUiProvider supportedChains={[mainnet, sepolia]}>
      {children}
    </DappUiProvider>
  )
}
```

---

## App Layout Usage

Add the wrapper to your root layout:

```tsx title="app/layout.tsx" showLineNumbers
import { DappUiWrapper } from "@/provider/dapp-ui.provider"

export default function RootLayout({ children }) {
  return (
    <html lang="en" suppressHydrationWarning>
      <body>
        <DappUiWrapper>{children}</DappUiWrapper>
      </body>
    </html>
  )
}
```

## Use Modular Hooks

You can use individual wallet hooksâ€¦

### Connect Wallet

```tsx {3} {6} {8} title="components/ConnectWallet.tsx" showLineNumbers
"use client"

import { useConnect } from "@web3-blocks/dapp-ui"

export function ConnectWallet() {
  const { connect, isConnecting, isWalletAvailable } = useConnect()

  if (!isWalletAvailable) return <p>No wallet detected</p>

  return (
    <button onClick={connect} disabled={isConnecting}>
      {isConnecting ? "Connecting..." : "Connect Wallet"}
    </button>
  )
}
```

### Account State

```tsx {3} {6} {8} title="components/Account.tsx" showLineNumbers
"use client"

import { useAccount } from "@web3-blocks/dapp-ui"

export function Account() {
  const { account, isConnected } = useAccount()

  if (!isConnected) return null

  return <p>{account?.address}</p>
}
```

### Disconnect

```tsx {3} {6} title="components/DisconnectButton.tsx" showLineNumbers
"use client"

import { useDisconnect } from "@web3-blocks/dapp-ui"

export function DisconnectButton() {
  const { disconnect, isDisconnecting } = useDisconnect()

  return (
    <button onClick={disconnect} disabled={isDisconnecting}>
      {isDisconnecting ? "Disconnecting..." : "Disconnect"}
    </button>
  )
}
```

## Optional: All-in-One Hook

Instead of multiple imports, use:

```tsx title="components/WalletPanel.tsx" showLineNumbers
"use client"

import { useEthereum } from "@web3-blocks/dapp-ui"

export function WalletPanel() {
  const {
    account: { account, isConnected },
    connect: { connect },
    disconnect: { disconnect },
  } = useEthereum()

  return (
    <div>
      {!isConnected && <button onClick={connect}>Connect</button>}
      {isConnected && <button onClick={disconnect}>Disconnect</button>}
      <pre>{JSON.stringify(account, null, 2)}</pre>
    </div>
  )
}
```
